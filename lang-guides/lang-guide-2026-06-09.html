package main

import (
    "fmt"
    "sort"
)

func main() {
    words := []string{"apple", "banana", "avocado", "blueberry", "apricot", "cherry"}

    groups := make(map[byte][]string)
    for _, w := range words {
        key := w[0]
        groups[key] = append(groups[key], w)
    }

    // Must sort keys manually — Go map iteration is random
    keys := make([]byte, 0, len(groups))
    for k := range groups {
        keys = append(keys, k)
    }
    sort.Slice(keys, func(i, j int) bool { return keys[i] < keys[j] })

    for _, k := range keys {
        fmt.Printf("%c: %v\n", k, groups[k])
    }
}

use std::collections::HashMap;

fn main() {
    let words = vec!["apple", "banana", "avocado", "blueberry", "apricot", "cherry"];

    let mut groups: HashMap<char, Vec<&str>> = HashMap::new();
    for w in &words {
        // entry() API avoids double lookup

              .or_insert_with(Vec::new)
              .push(w);
    }

    let mut keys: Vec<&char> = groups.keys().collect();

    }
}

list as stack (Python), slice (Go), Array (TS), Vec/VecDeque (Rust), stack/queue/deque (C++). Queue with two stacks: push to in_stack, pop by transferring to out_stack when empty. Amortized O(1) for all operations.

class QueueFromStacks:
    def __init__(self):
        self._in = []   # push here
        self._out = []  # pop from here

    def enqueue(self, val):
        self._in.append(val)

    def dequeue(self):
        if not self._out:
            while self._in:
                self._out.append(self._in.pop())
        if not self._out:
            raise IndexError("empty queue")
        return self._out.pop()

package main

import "fmt"

type QueueFromStacks struct {
    in, out []int
}

func (q *QueueFromStacks) Enqueue(val int) {
    q.in = append(q.in, val)
}

func (q *QueueFromStacks) Dequeue() int {
    if len(q.out) == 0 {
        for len(q.in) > 0 {
            top := q.in[len(q.in)-1]
            q.in = q.in[:len(q.in)-1]
            q.out = append(q.out, top)
        }
    }
    val := q.out[len(q.out)-1]
    q.out = q.out[:len(q.out)-1]
    return val
}

    q := &QueueFromStacks{}

        q.Enqueue(x)

class QueueFromStacks<T> {
    private inStack: T[] = [];
    private outStack: T[] = [];

    enqueue(val: T): void {
        this.inStack.push(val);
    }

    dequeue(): T {
        if (this.outStack.length === 0) {
            while (this.inStack.length > 0) {
                this.outStack.push(this.inStack.pop()!);
            }

        if (this.outStack.length === 0) throw new Error("empty queue");
        return this.outStack.pop()!;

}

Rust
struct QueueFromStacks<T> {
    in_stack: Vec<T>,
    out_stack: Vec<T>,
}

impl<T> QueueFromStacks<T> {
    fn new() -> Self {
        Self { in_stack: Vec::new(), out_stack: Vec::new() }
    }

    fn enqueue(&mut self, val: T) {
        self.in_stack.push(val);
    }

    fn dequeue(&mut self) -> Option<T> {
        if self.out_stack.is_empty() {
            while let Some(val) = self.in_stack.pop() {
                self.out_stack.push(val);
            }
        }
        self.out_stack.pop()
    }
}

fn main() {
    let mut q = QueueFromStacks::new();
    for x in [1, 2, 3] { q.enqueue(x); }
    println!("{} {}", q.dequeue().unwrap(), q.dequeue().unwrap());
    q.enqueue(4);
    println!("{} {}", q.dequeue().unwrap(), q.dequeue().unwrap());
}

#include <iostream>
#include <stack>
#include <stdexcept>
using namespace std;

template<typename T>
class QueueFromStacks {
    stack<T> in_, out_;
public:
    void enqueue(T val) { in_.push(val); }

    T dequeue() {
        if (out_.empty()) {
            while (!in_.empty()) {
                out_.push(in_.top());
                in_.pop();
            }
        }
        if (out_.empty()) throw runtime_error("empty queue");
        T val = out_.top();
        out_.pop();
        return val;
    }
};

Assertion Patterns & Custom Matchers Testing 2 days since · rep 0
pytest assert (Python), testify (Go), expect (TS/Jest), assert_eq! (Rust), EXPECT_EQ (C++/gtest). For floating point comparison, always use an epsilon/tolerance. Custom assertions improve test readability.
Python
import math

def is_sorted(lst: list) -> bool:
    return all(lst[i] <= lst[i+1] for i in range(len(lst)-1))

def assert_sorted(lst: list, msg: str = ""):
    if not is_sorted(lst):
        # Find first violation
        for i in range(len(lst)-1):
            if lst[i] > lst[i+1]:
                raise AssertionError(
                    f"Not sorted at index {i}: {lst[i]} > {lst[i+1]}" +
                    (f" ({msg})" if msg else ""))

assert_sorted([1, 2, 3, 4, 5])
print("sorted: PASS")

# Float comparison with tolerance
assert math.isclose(0.1 + 0.2, 0.3, rel_tol=1e-9)
print("float: PASS")

fn is_sorted(s: &[i32]) -> Result<(), (usize, i32, i32)> {
    for i in 0..s.len().saturating_sub(1) {
        if s[i] > s[i + 1] {
            return Err((i, s[i], s[i + 1]));
        }
    }
    Ok(())
}

        panic!("Not sorted at index {}: {} > {}", idx, a, b);

async def fetch(name: str, delay: float, fail: bool = False) -> str:
    await asyncio.sleep(delay)
    if fail:
        raise ValueError(f"{name} failed")

async def first_success(tasks):
    """Return the first successful result, cancel the rest."""
    done, pending = await asyncio.wait(tasks, return_when=asyncio.FIRST_COMPLETED)
    for t in pending:
        t.cancel()
    for t in done:
        if not t.exception():
            return t.result()
    raise RuntimeError("All tasks failed")

    "time"

func fetch(ctx context.Context, name string, delay time.Duration, fail bool) (string, error) {
    select {
    case <-time.After(delay):
        if fail {
            return "", fmt.Errorf("%s failed", name)
        }
        return fmt.Sprintf("%s: OK (%v)", name, delay), nil
    case <-ctx.Done():
        return "", ctx.Err()
    }
}

func firstSuccess(fns []func(context.Context) (string, error)) (string, error) {
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    type result struct {
        val string
        err error
    }
    ch := make(chan result, len(fns))
    for _, fn := range fns {
        fn := fn
        go func() { v, e := fn(ctx); ch <- result{v, e} }()
    }
    for range fns {
        r := <-ch
        if r.err == nil {
            return r.val, nil // cancel() deferred — stops others
        }
    }
    return "", errors.New("all tasks failed")
}

Pitfalls to Remember:
  • Python: Forgetting await silently returns a coroutine object instead of the result.
  • Go: Goroutine leak: always use context.WithCancel so goroutines check ctx.Done() and exit.
  • TypeScript: Unhandled promise rejections crash Node.js — always .catch() or use try/catch with await.
  • Rust: Dropping a JoinHandle without .await detaches the task — it keeps running in the background.
  • C++: std::async with launch::async blocks on destructor — the future’s destructor waits for the thread.

New Concepts — Week 6: Advanced Patterns

Four new concepts exploring cross-language boundaries, architecture patterns, event-driven design, and deployment configuration.

FFI & Interop Advanced Patterns
Sometimes you need C library performance or must call system libraries not available in your language. FFI (Foreign Function Interface) bridges that gap — but it introduces unsafe boundaries that require careful memory management.

Core Idea

Every language has a way to call C functions. The C ABI (Application Binary Interface) is the universal lingua franca of system programming. FFI lets you call C functions, pass data across the boundary, and receive callbacks — but you must manage memory ownership carefully because the GC/borrow checker cannot see across FFI boundaries.

Language Comparison: Python: ctypes is built-in but requires manual type declarations. cffi parses C headers for you. Wrong argtypes causes segfaults with no helpful error.

Go: cgo embeds C code in comments directly above import "C". It is convenient but adds ~100us overhead per call and complicates cross-compilation.

TypeScript: No native FFI in browsers. Node.js uses ffi-napi or N-API. WebAssembly is the cross-platform FFI story — compile C to WASM and call from JS.

Rust: FFI requires unsafe blocks. #[repr(C)] ensures struct layout compatibility. The bindgen crate auto-generates Rust bindings from C headers.

C++: Already speaks C natively via extern "C". The main challenge is preventing C++ name mangling when exposing functions to other languages.
Common Mistakes:
  • Python: Forgetting to set argtypes/restype on ctypes functions — this causes silent memory corruption or segfaults.
  • Go: cgo overhead is ~100us per call — do not use it in tight loops. Batch operations instead.
  • TypeScript: WASM memory is separate from JS heap — you must copy data in/out via ArrayBuffer views.
  • Rust: Forgetting #[repr(C)] on structs passed across FFI — Rust’s default layout is unspecified and may differ from C.
  • C++: Not using extern "C" causes name mangling, making functions invisible to other languages.

Dependency Injection Advanced Patterns
DI enables testability, loose coupling, and swappable implementations. Instead of a class creating its own dependencies, they are provided from outside — making it trivial to swap a real database for a mock in tests.

Core Idea

Dependency Injection is a form of Inversion of Control: instead of new Database() inside your service, you pass the database as a constructor parameter. This lets you inject different implementations (real DB, mock DB, in-memory DB) without changing the service code.

Event Systems & Reactive Patterns Advanced Patterns
Event-driven architecture decouples producers from consumers, enabling scalable async systems. Instead of calling functions directly, you emit events and let any number of listeners respond independently.

Core Idea

The pub/sub pattern: publishers emit named events, subscribers register handlers. This decouples components — the publisher does not know who is listening. Each language has a different native mechanism: Python uses dict-of-callbacks (no built-in EventEmitter), Go uses channels, TypeScript has EventEmitter/EventTarget, Rust uses tokio channels, and C++ uses function pointers or std::function.

Example A — Simple typed EventEmitter

Python
from typing import Callable, Any
from collections import defaultdict

class EventEmitter:
    def __init__(self):
        self._listeners: dict[str, list[Callable]] = defaultdict(list)

    def on(self, event: str, handler: Callable) -> None:
        self._listeners[event].append(handler)

    def off(self, event: str, handler: Callable) -> None:
        self._listeners[event].remove(handler)

    def emit(self, event: str, *args: Any) -> None:
        for handler in self._listeners[event]:
            handler(*args)

Language Comparison: Python: pydantic-settings automatically reads env vars with validation and type coercion. For simpler projects, os.environ + dataclass with __post_init__ works well.

Go: viper supports env vars, config files (YAML/JSON/TOML), and command-line flags with a unified API. envconfig maps struct tags to env vars.

TypeScript: dotenv loads .env files. convict or zod for schema validation. process.env values are always strings — always parse.

Rust: config crate supports layered config (env + file + defaults). envy deserializes env vars into structs via serde. dotenvy loads .env files.

C++: No standard config library. getenv() for env vars. yaml-cpp or nlohmann/json for config files. Build your own validation layer.
Common Mistakes:
  • All languages: Logging config values that contain secrets (passwords, API keys) — redact sensitive fields in logs.
  • Python: Using os.environ["KEY"] throws KeyError if missing — use .get() with a default instead.
  • Go: os.Getenv() returns empty string for both “unset” and “set to empty” — use os.LookupEnv() to distinguish.
  • TypeScript: process.env.PORT is string | undefined — always handle the undefined case before parsing.
  • Rust: env::var() returns Result — unwrapping without a default panics at runtime if the var is missing.
  • C++: getenv() returns nullptr for missing vars — always null-check before using the result.
12-Factor Config Rules:
  • Never hardcode connection strings, API keys, or environment-specific values.
  • Validate at startup — fail fast with a clear error message rather than crashing at first use.
  • Provide sensible defaults for development, require explicit values in production.
  • Use .env files for local development, real env vars in production. Never commit .env to git.
