1. Arrays, Slices & Vectors Collections rep 3 • 15 days ago
Due today
Quick Recall: Core sequential containers differ by ownership/mutability. Python list is dynamic, Go slice has len+cap, TS Array is reference, Rust Vec owns heap data, C++ vector manages memory via RAII.

Key Patterns

Python 
# Slice every 2nd element, reverse, list comprehension
nums = [1, 2, 3, 4, 5, 6, 7, 8]
evens = nums[::2]        # [1, 3, 5, 7]
rev = nums[::-1]          # [8, 7, 6, 5, 4, 3, 2, 1]
squares = [x**2 for x in nums]  # [1, 4, 9, 16, 25, 36, 49, 64]
Go 
// append may reallocate; len vs cap distinction
s := []int{1, 2, 3}
s = append(s, 4, 5)         // len=5, cap may be 6
sub := s[1:3]               // [2 3] shares underlying array
cp := make([]int, len(s))
copy(cp, s)                 // independent copy
TypeScript 
// Array is reference type; sort is lexicographic by default
const nums: number[] = [10, 2, 30, 4];
const mapped = nums.map(x => x * 2);      // [20, 4, 60, 8]
const sorted = [...nums].sort((a, b) => a - b);  // [2, 4, 10, 30]
const flat = [[1, 2], [3, 4]].flat();     // [1, 2, 3, 4]
Rust 
// Vec owns heap data; iter/filter/collect chain
let mut v = vec![1, 2, 3, 4, 5];
v.push(6);
let evens: Vec<_> = v.iter().filter(|&&x| x % 2 == 0).collect();
// evens = [2, 4, 6]
let slice = &v[1..3];  // borrows [2, 3] -- cannot mutate v while borrowed
C++ 
// vector manages memory via RAII; iterator invalidation on push_back
#include <vector>
#include <algorithm>
std::vector<int> v = {3, 1, 4, 1, 5};
v.push_back(9);
std::sort(v.begin(), v.end());        // {1, 1, 3, 4, 5, 9}
v.erase(std::unique(v.begin(), v.end()), v.end()); // {1, 3, 4, 5, 9}
Pitfalls to Remember
  • Python: Negative indexing wraps silently — nums[-1] is last element, not an error
  • Go: append may reallocate; sub-slices share underlying array until append triggers growth
  • TS: [10, 2, 30].sort() gives [10, 2, 30] (lexicographic!) — always pass comparator
  • Rust: Cannot push to a Vec while a slice borrows it (borrow checker)
  • C++: push_back invalidates iterators if reallocation occurs

Pitfalls to Remember
  • Python: GIL does NOT make your code thread-safe — compound operations still need locks

  • C++: Forgetting to scope lock_guard — lock held longer than necessary

func IsPalindrome(s string) bool {
    cleaned := strings.ToLower(strings.ReplaceAll(s, " ", ""))
    runes := []rune(cleaned)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        if runes[i] != runes[j] { return false }
    }
    return true
}

func TestIsPalindrome(t *testing.T) {
    tests := []struct {
        input string; want bool
    }{
        {"racecar", true}, {"hello", false},
        {"A man a plan a canal Panama", true},
        {"", true}, {"a", true},
    }
    for _, tc := range tests {
        t.Run(tc.input, func(t *testing.T) {
            if got := IsPalindrome(tc.input); got != tc.want {
                t.Errorf("IsPalindrome(%q) = %v, want %v", tc.input, got, tc.want)
            }
        })
    }
}

Rust 
// Custom global allocator for tracking
use std::alloc::{GlobalAlloc, Layout, System};
use std::sync::atomic::{AtomicUsize, Ordering};

struct TrackingAllocator;
static ALLOCATED: AtomicUsize = AtomicUsize::new(0);

unsafe impl GlobalAlloc for TrackingAllocator {
    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
        ALLOCATED.fetch_add(layout.size(), Ordering::Relaxed);
        unsafe { System.alloc(layout) }
    }
    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
        ALLOCATED.fetch_sub(layout.size(), Ordering::Relaxed);
        unsafe { System.dealloc(ptr, layout) }
    }
}

#[global_allocator]
static A: TrackingAllocator = TrackingAllocator;

fn main() {
    let before = ALLOCATED.load(Ordering::Relaxed);
    let data: Vec<i32> = (0..1_000_000).collect();
    let after = ALLOCATED.load(Ordering::Relaxed);

Example 1: Process a File Line by Line Without Loading Into Memory

The fundamental pattern: read one chunk at a time, process it, discard it. Memory stays constant regardless of file size.

from typing import Iterator

def read_lines(path: str) -> Iterator[str]:
    """Generator that yields lines lazily -- constant memory."""
    with open(path, "r") as f:
        for line in f:
            yield line.strip()

def filter_errors(lines: Iterator[str]) -> Iterator[str]:
    """Filter stage: only pass through ERROR lines."""
    for line in lines:
        if "ERROR" in line:
            yield line

    """Transform stage: extract just the message part."""
    for line in lines:
        parts = line.split(" ", 3)
        if len(parts) >= 4:
            yield parts[3]

When producers are faster than consumers, backpressure prevents memory from growing unbounded. Each language handles this differently.

#include <condition_variable>
#include <iostream>
#include <mutex>
#include <queue>
#include <thread>

template <typename T>
class BoundedQueue {
    std::queue<T> q_;
    size_t max_size_;
    std::mutex mtx_;
    std::condition_variable not_full_, not_empty_;
    bool done_ = false;

public:
    explicit BoundedQueue(size_t max) : max_size_(max) {}

    void push(T val) {
        std::unique_lock lk(mtx_);
        not_full_.wait(lk, [&] { return q_.size() < max_size_; });
        q_.push(std::move(val));
        not_empty_.notify_one();
    }

    bool pop(T& val) {
        std::unique_lock lk(mtx_);
        not_empty_.wait(lk, [&] { return !q_.empty() || done_; });
        if (q_.empty()) return false;
        val = std::move(q_.front());
        q_.pop();
        not_full_.notify_one();
        return true;
    }

    void close() {
        std::lock_guard lk(mtx_);
        done_ = true;
        not_empty_.notify_all();
    }
};

int main() {
    BoundedQueue<int> bq(5); // backpressure at 5 items

    std::thread producer([&] {
        for (int i = 0; i < 100; ++i) {
            bq.push(i);
            std::cout << "Produced: " << i << "\n";
        }
        bq.close();
    });

    std::thread consumer([&] {
        int val;
        while (bq.pop(val)) {
            std::this_thread::sleep_for(std::chrono::milliseconds(10));
            std::cout << "Consumed: " << val << "\n";
        }
    });

Language Comparison: Python uses generator pipelines (lazy, pull-based). Go uses channel pipelines (concurrent, push-based with backpressure via blocking). Rust combines iterator chains (zero-cost, pull-based) with async streams for I/O. C++20 ranges bring lazy evaluation to the STL. TS Node streams are push-based with built-in backpressure.
Common Mistakes
  • Python: Materializing entire generator with list() — defeats the purpose of lazy evaluation
  • Go: Not closing channels — causes goroutine leaks (goroutines block forever on range)
  • TS: Not handling stream errors or backpressure — data loss or OOM
  • Rust: Calling .collect() on intermediate results instead of chaining iterators
  • C++: Mixing istream_iterator with views:: incorrectly — input iterators are single-pass
Key Rules
  • Process data lazily — only materialize when you need the final result
  • Always close channels/streams when the producer is done
  • Use bounded buffers to implement backpressure and prevent OOM

fn validate_host(s: &str) -> Result<String, String> {
    if s.is_empty() { Err("host is required".into()) }
    else { Ok(s.to_string()) }
}

fn validate_port(p: i32) -> Result<u16, String> {
    if p < 1 || p > 65535 { Err(format!("port must be 1-65535, got {}", p)) }
    else { Ok(p as u16) }
}

fn validate_timeout(t: i64) -> Result<u64, String> {
    if t <= 0 { Err(format!("timeout must be positive, got {}", t)) }
    else { Ok(t as u64) }
}

// Collect all errors instead of short-circuiting
fn validate_config(host: &str, port: i32, timeout: i64) -> Result<Config, Vec<String>> {
    let results = (
        validate_host(host),
        validate_port(port),
        validate_timeout(timeout),
    );

    let mut errors = Vec::new();
    if let Err(e) = &results.0 { errors.push(e.clone()); }
    if let Err(e) = &results.1 { errors.push(e.clone()); }
    if let Err(e) = &results.2 { errors.push(e.clone()); }

    if !errors.is_empty() {
        return Err(errors);
    }

    Ok(Config {
        host: results.0.unwrap(),
        port: results.1.unwrap(),
        timeout_ms: results.2.unwrap(),
    })
}

fn main() {
    match validate_config("", -1, 0) {
        Ok(c) => println!("Config: {:?}", c),
        Err(errs) => println!("Errors: {:?}", errs),
    }
}

#include <expected>
#include <iostream>
#include <string>
#include <vector>

struct Config {
    std::string host;
    int port;
    double timeout;
};

using Errors = std::vector<std::string>;
using Validated = std::expected<Config, Errors>;

Validated validate_config(const std::string& host, int port, double timeout) {
    Errors errors;

    if (host.empty())
        errors.push_back("host is required");
    if (port < 1 || port > 65535)
        errors.push_back("port must be 1-65535, got " + std::to_string(port));
    if (timeout <= 0)
        errors.push_back("timeout must be positive, got " + std::to_string(timeout));

    if (!errors.empty())
        return std::unexpected(errors);

    return Config{host, port, timeout};
}

int main() {
    auto result = validate_config("", -1, 0);

        std::cout << "Errors:\n";
        for (const auto& e : result.error())
            std::cout << "  - " << e << "\n";
    }
}

Language Comparison: Rust has the best built-in support with ? and Result method chains. Go deliberately avoids monadic patterns — explicit if err != nil is idiomatic. Python and TS need third-party libraries (returns, fp-ts) for true railway-oriented programming. C++23's std::expected finally brings Result-like semantics to C++.
Common Mistakes
  • Python: Catching too-broad exceptions (except Exception) instead of using Result types for expected failures
  • Go: Ignoring returned errors with _ = fn() — the blank identifier anti-pattern
  • TS: Using try/catch everywhere instead of Result when composing — exceptions are for unexpected failures
  • Rust: Using .unwrap() in library code instead of propagating with ?
  • C++: Not checking expected before accessing value — undefined behavior if error
Key Rules
  • Use Result/Expected types for expected failures; reserve exceptions/panics for unexpected ones
  • Short-circuit for single-error pipelines; collect for multi-field validation
  • Wrap errors with context at each layer — "what was I trying to do when this failed?"
