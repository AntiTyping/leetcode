Dunder Methods — __iter__, __next__, __call__

OOP • 9 days ago • rep 1

Quick Reminder

__iter__ returns the iterator object (often self), __next__ returns the next value or raises StopIteration, __call__ makes instances callable like functions.

Key Pattern

class Countdown:
    def __init__(self, start):
        self.current = start
    def __iter__(self):
        return self
    def __next__(self):
        if self.current <= 0:
            raise StopIteration
        self.current -= 1
        return self.current + 1

class Multiplier:
    def __init__(self, factor):
        self.factor = factor
    def __call__(self, x):
        return x * self.factor

double = Multiplier(2)
print(double(5))  # 10
10

Descriptors

OOP • 9 days ago • rep 1

Quick Reminder

Descriptors implement __get__, __set__, __delete__ to intercept attribute access. Data descriptors (with __set__) take priority over instance __dict__.

Key Pattern

class Validated:
    def __init__(self, min_val, max_val):
        self.min_val = min_val
        self.max_val = max_val
    def __set_name__(self, owner, name):
        self.name = name
    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        return obj.__dict__.get(self.name)
    def __set__(self, obj, value):
        if not self.min_val <= value <= self.max_val:
            raise ValueError(f"{self.name} must be between {self.min_val} and {self.max_val}")
        obj.__dict__[self.name] = value

Exception Hierarchy & Custom Exceptions

Error Handling • 9 days ago • rep 1

Quick Reminder

All exceptions inherit from BaseException. Use Exception for application errors. Create custom hierarchies with specific __init__ for context. Use exception groups (ExceptionGroup) for multiple errors.

Key Pattern

class AppError(Exception):
    """Base for all app errors."""

class ValidationError(AppError):
    def __init__(self, field, message):
        self.field = field
        super().__init__(f"{field}: {message}")

class NotFoundError(AppError):
    def __init__(self, resource, id_):
        self.resource = resource
        self.id = id_
        super().__init__(f"{resource} {id_} not found")

try:
    raise ValidationError("email", "invalid format")
except AppError as e:
    print(type(e).__name__, e)  # ValidationError email: invalid format
ValidationError email: invalid format

Design Patterns in Python

Testing & Packaging • 7 days overdue • rep 0

Quick Reminder

Python’s duck typing makes some patterns simpler — Strategy is just a function, Singleton uses __new__ or a module, Factory uses classmethods, Observer uses callbacks or built-in patterns.

Key Pattern

# Strategy pattern — just pass a function
def process(data, strategy):
    return strategy(data)

result = process([3, 1, 2], sorted)       # [1, 2, 3]
result = process([3, 1, 2], lambda d: d[::-1])  # [2, 1, 3]

# Observer
class EventEmitter:
    def __init__(self):
        self._listeners = {}
    def on(self, event, callback):
        self._listeners.setdefault(event, []).append(callback)
    def emit(self, event, *args):
        for cb in self._listeners.get(event, []):
            cb(*args)

hashlib & secrets — Hashing & Secure Random

Security & Cryptography

Every application that handles passwords, verifies file integrity, or generates tokens needs cryptographic hashing and secure randomness. Python’s hashlib and secrets modules are the correct tools — never use random for security.

Example 1 — Basic hashing

import hashlib

# Hash a string (must encode to bytes first)
digest = hashlib.sha256(b"hello world").hexdigest()
print(digest)  # a591a6d40bf420404a011733cfb7b190d62c65bf0bcda32b57b277d9ad9f146e
print(len(digest))  # 64 hex chars = 32 bytes

# MD5 (not for security, but useful for checksums)
print(hashlib.md5(b"hello").hexdigest())  # 5d41402abc4b2a76b9719d911017c592
a591a6d40bf420404a011733cfb7b190d62c65bf0bcda32b57b277d9ad9f146e 64 5d41402abc4b2a76b9719d911017c592

Example 2 — Incremental hashing (large files)

import hashlib

def file_hash(path, algo="sha256", chunk_size=8192):
    h = hashlib.new(algo)
    with open(path, "rb") as f:
        while chunk := f.read(chunk_size):
            h.update(chunk)
    return h.hexdigest()

# Equivalent to hashing the whole file at once, but memory-efficient
# file_hash("large_file.bin")

Example 3 — secrets module for secure tokens

import secrets

# Secure random token (URL-safe)
token = secrets.token_urlsafe(32)
print(token)  # e.g., "dGhpcyBpcyBhIHNlY3VyZSB0b2tlbg..."
print(len(token))  # ~43 chars

# Secure random hex
hex_token = secrets.token_hex(16)
print(hex_token)  # e.g., "a3f2b8c9d1e4f5a6b7c8d9e0f1a2b3c4"

# Secure random bytes
raw = secrets.token_bytes(16)
print(len(raw))  # 16

# Secure random integer
print(secrets.randbelow(100))  # 0-99, cryptographically secure

# Compare strings in constant time (prevents timing attacks)
print(secrets.compare_digest("abc", "abc"))  # True
print(secrets.compare_digest("abc", "abd"))  # False

Example 4 — Password hashing with PBKDF2

import hashlib
import secrets

def hash_password(password: str) -> str:
    salt = secrets.token_hex(16)
    dk = hashlib.pbkdf2_hmac("sha256", password.encode(), salt.encode(), iterations=600_000)
    return f"{salt}${dk.hex()}"

def verify_password(password: str, stored: str) -> bool:
    salt, hash_hex = stored.split("$")
    dk = hashlib.pbkdf2_hmac("sha256", password.encode(), salt.encode(), iterations=600_000)
    return secrets.compare_digest(dk.hex(), hash_hex)

stored = hash_password("mypassword")
print(stored)  # "a1b2c3...$d4e5f6..."
print(verify_password("mypassword", stored))   # True
print(verify_password("wrongpassword", stored)) # False
(salt)$(hash) True False

Example 5 — HMAC for message authentication

import hmac
import hashlib

key = b"my-secret-key"
message = b"transfer $100 to Alice"

# Create HMAC
signature = hmac.new(key, message, hashlib.sha256).hexdigest()
print(signature)  # consistent hash that requires the key

# Verify — use hmac.compare_digest (constant time)
received_sig = signature  # pretend we received this
is_valid = hmac.compare_digest(
    hmac.new(key, message, hashlib.sha256).hexdigest(),
    received_sig
)
print(is_valid)  # True

Common Mistakes

Don’t
random.randint() for tokens
# Predictable, not cryptographically secure

hashlib.sha256(password) alone
# Vulnerable to rainbow tables

hash_a == hash_b
# Timing attack vulnerability
Do
secrets.token_urlsafe() or token_hex()
# Cryptographically secure randomness

pbkdf2_hmac with random salt + high iterations
# Resistant to rainbow tables

secrets.compare_digest(a, b)
# Constant-time comparison

Key Rules

  • hashlib for one-way hashing (integrity, checksums). secrets for randomness that must be unpredictable.
  • Never use MD5 or SHA-1 for security (use SHA-256+). MD5 is fine for non-security checksums.
  • Always salt passwords. Use pbkdf2_hmac with ≥600,000 iterations (OWASP recommendation).

pickle & shelve — Object Serialization

Serialization

pickle serializes arbitrary Python objects to bytes — useful for caching, IPC, and ML model storage. shelve provides a dict-like persistent store backed by pickle. Both have serious security implications.

Example 1 — Basic pickling

import pickle

data = {"name": "Alice", "scores": [95, 87, 92], "active": True}

# Serialize to bytes
pickled = pickle.dumps(data)
print(type(pickled), len(pickled))  # <class 'bytes'> ~65

# Deserialize
restored = pickle.loads(pickled)
print(restored)  # {'name': 'Alice', 'scores': [95, 87, 92], 'active': True}
print(data == restored)  # True

Example 2 — Pickle to file

import pickle

class Model:
    def __init__(self, weights):
        self.weights = weights
    def predict(self, x):
        return sum(w * xi for w, xi in zip(self.weights, x))

model = Model([0.5, -0.3, 0.8])

# Save
with open("/tmp/model.pkl", "wb") as f:
    pickle.dump(model, f)

# Load
with open("/tmp/model.pkl", "rb") as f:
    loaded = pickle.load(f)

print(loaded.predict([1, 2, 3]))  # 0.5 - 0.6 + 2.4 = 2.3
print(type(loaded))  # <class '__main__.Model'>
2.3 <class '__main__.Model'>

Example 3 — Customizing pickling

import pickle

class Connection:
    def __init__(self, host, port):
        self.host = host
        self.port = port
        self.socket = f"<socket to {host}:{port}>"  # unpicklable resource

    def __getstate__(self):
        state = self.__dict__.copy()
        del state["socket"]  # remove unpicklable
        return state

    def __setstate__(self, state):
        self.__dict__.update(state)
        self.socket = f"<socket to {self.host}:{self.port}>"  # reconnect

conn = Connection("localhost", 5432)
restored = pickle.loads(pickle.dumps(conn))
print(restored.host, restored.socket)
# localhost <socket to localhost:5432>
localhost <socket to localhost:5432>

Example 4 — shelve for persistent dict

import shelve

# shelve works like a persistent dict
with shelve.open("/tmp/mydata") as db:
    db["users"] = ["Alice", "Bob"]
    db["config"] = {"theme": "dark", "lang": "en"}

# Later, in another session:
with shelve.open("/tmp/mydata") as db:
    print(list(db.keys()))   # ['users', 'config']
    print(db["users"])       # ['Alice', 'Bob']

    # Gotcha: mutable values need reassignment
    users = db["users"]
    users.append("Charlie")
    db["users"] = users      # must reassign!
    print(db["users"])       # ['Alice', 'Bob', 'Charlie']

Common Mistakes

Don’t
# Loading pickle from untrusted sources
pickle.loads(data_from_internet)
# Could run arbitrary code!

# Mutating shelve values in place
db["items"].append("x")
# Change is LOST

# Pickle for long-term storage
# Class changes break unpickling
Do
# Use JSON for untrusted data
json.loads(data_from_internet)
# Safe — no code execution

# Reassign after mutation
items = db["items"]
items.append("x")
db["items"] = items

# JSON, database, or protobuf for durable storage

Key Rules

  • NEVER unpickle untrusted data — it’s a remote code execution vector.
  • Use __getstate__/__setstate__ to customize serialization (skip sockets, locks, etc.).
  • shelve is dict-like persistence for small data — not a database replacement.

Numeric Precision & Data

decimal.Decimal & fractions.Fraction — Exact Arithmetic

Numeric Precision

Floating-point arithmetic produces surprising results (0.1 + 0.2 != 0.3). Financial calculations, scientific computation, and any domain requiring exact decimal results needs Decimal. Fraction handles exact rational arithmetic.

Example 1 — The floating-point problem

from decimal import Decimal

# The classic float problem
print(0.1 + 0.2)           # 0.30000000000000004
print(0.1 + 0.2 == 0.3)    # False

# Decimal fixes this
print(Decimal("0.1") + Decimal("0.2"))          # 0.3
print(Decimal("0.1") + Decimal("0.2") == Decimal("0.3"))  # True
0.30000000000000004 False 0.3 True

Example 2 — Decimal precision and rounding

from decimal import Decimal, ROUND_HALF_UP, ROUND_DOWN, getcontext

# Always create from strings (not floats!)
price = Decimal("19.99")
tax_rate = Decimal("0.0875")
tax = price * tax_rate
print(tax)  # 1.749125

# Round to cents
print(tax.quantize(Decimal("0.01")))                    # 1.75
print(tax.quantize(Decimal("0.01"), ROUND_HALF_UP))     # 1.75
print(tax.quantize(Decimal("0.01"), ROUND_DOWN))        # 1.74

# Set global precision
getcontext().prec = 50
print(Decimal(1) / Decimal(7))  # 0.14285714285714285714285714285714285714285714285714

Example 3 — Financial calculation

from decimal import Decimal, ROUND_HALF_UP

def calculate_total(items):
    """Calculate order total with tax, rounding each line item."""
    subtotal = Decimal("0")
    for name, price, qty in items:
        line = Decimal(str(price)) * qty
        subtotal += line
    tax = (subtotal * Decimal("0.08")).quantize(Decimal("0.01"), ROUND_HALF_UP)
    return subtotal, tax, subtotal + tax

items = [("Widget", 9.99, 3), ("Gadget", 24.50, 2)]
sub, tax, total = calculate_total(items)
print(f"Subtotal: ${sub}, Tax: ${tax}, Total: ${total}")
# Subtotal: $78.97, Tax: $6.32, Total: $85.29
Subtotal: $78.97, Tax: $6.32, Total: $85.29

Example 4 — Fractions

from fractions import Fraction

# Create from various inputs
print(Fraction(1, 3))          # 1/3
print(Fraction("3.14"))        # 157/50
print(Fraction(0.1))           # 3602879701896397/36028797018963968 (exact float rep)
print(Fraction("0.1"))         # 1/10 (from string is exact)

# Arithmetic is exact
a = Fraction(1, 3)
b = Fraction(1, 6)
print(a + b)      # 1/2
print(a * b)      # 1/18
print(a - b)      # 1/6

# Convert back
print(float(Fraction(1, 3)))   # 0.3333333333333333

# Limit denominator (approximate)
pi_approx = Fraction("3.14159265358979")
print(pi_approx.limit_denominator(100))   # 311/99
print(pi_approx.limit_denominator(1000))  # 355/113

Common Mistakes

Don’t
# Creating Decimal from float
Decimal(0.1)
# 0.10000000000000000555...

# Mixing Decimal and float
Decimal("1.5") + 0.1
# TypeError!

# Using round() for financial rounding
round(2.5)  # 2 (banker's rounding)
Do
# Always create from string
Decimal("0.1")
# Exactly 0.1

# Keep everything as Decimal
Decimal("1.5") + Decimal("0.1")

# Use .quantize() with explicit mode
tax.quantize(Decimal("0.01"), ROUND_HALF_UP)

Key Rules

  • Use Decimal for money, finance, and anywhere exact decimal representation matters. Create from strings.
  • Use Fraction when you need exact rational arithmetic (recipe scaling, music timing, etc.).
  • Float is fine for science/graphics where small errors are acceptable and speed matters.

statistics Module

Data Analysis

Python’s statistics module provides production-ready statistical functions without needing NumPy. Perfect for quick calculations, data validation, and lightweight analytics.

Example 1 — Central tendency

import statistics

data = [4, 8, 15, 16, 23, 42]

print(statistics.mean(data))      # 18
print(statistics.median(data))    # 15.5  (average of middle two)
print(statistics.mode([1, 2, 2, 3, 3, 3]))  # 3
print(statistics.multimode([1, 1, 2, 2, 3]))  # [1, 2]

# Geometric & harmonic means
print(statistics.geometric_mean([1, 2, 4, 8]))  # 2.8284...
print(statistics.harmonic_mean([40, 60]))         # 48.0
18 15.5 3 [1, 2] 2.8284271247461903 48.0

Example 2 — Spread measures

import statistics

data = [2, 4, 4, 4, 5, 5, 7, 9]

print(statistics.variance(data))    # 4.571... (sample variance)
print(statistics.pvariance(data))   # 4.0     (population variance)
print(statistics.stdev(data))       # 2.138... (sample std dev)
print(statistics.pstdev(data))      # 2.0     (population std dev)

# Quantiles
print(statistics.quantiles(data))              # [3.5, 4.5, 6.5] (quartiles)
print(statistics.quantiles(data, n=10))        # deciles
print(statistics.quantiles(range(100), n=4))   # [24.75, 49.5, 74.25]

Example 3 — Linear regression and correlation

import statistics

# Study hours vs exam scores
hours = [1, 2, 3, 4, 5, 6, 7, 8]
scores = [52, 58, 65, 68, 73, 79, 84, 90]

# Correlation coefficient
r = statistics.correlation(hours, scores)
print(f"Correlation: {r:.4f}")  # ~0.9944 (strong positive)

# Linear regression
slope, intercept = statistics.linear_regression(hours, scores)
print(f"Score = {slope:.2f} * hours + {intercept:.2f}")
# Score ≈ 5.43 * hours + 48.00

# Predict for 10 hours
predicted = slope * 10 + intercept
print(f"Predicted score for 10 hours: {predicted:.1f}")  # ~102.3

Example 4 — NormalDist for probability

from statistics import NormalDist

# IQ scores: mean=100, stdev=15
iq = NormalDist(mu=100, sigma=15)

print(f"P(IQ < 130): {iq.cdf(130):.4f}")         # ~0.9772
print(f"P(IQ > 130): {1 - iq.cdf(130):.4f}")      # ~0.0228
print(f"P(85 < IQ < 115): {iq.cdf(115) - iq.cdf(85):.4f}")  # ~0.6827

# Quantiles — what IQ is the 95th percentile?
print(f"95th percentile: {iq.inv_cdf(0.95):.1f}")  # ~124.7

# Overlap between two distributions
exam_a = NormalDist(75, 10)
exam_b = NormalDist(80, 12)
print(f"Overlap: {exam_a.overlap(exam_b):.4f}")    # ~0.8037

Common Mistakes

Don’t
# mode() on data with no repeats (Python < 3.8)
statistics.mode([1, 2, 3])
# StatisticsError!

# Confusing sample vs population variance
statistics.variance(population_data)
# Wrong divisor (n-1 vs n)

# mean() on empty sequence
statistics.mean([])
# StatisticsError!
Do
# Use multimode() — always returns a list
statistics.multimode([1, 2, 3])
# [1, 2, 3]

# pvariance/pstdev for full population
statistics.pvariance(population_data)

# Check length first
if data:
    statistics.mean(data)

Key Rules

  • Use statistics for stdlib-only analysis. For heavy computation, use NumPy/pandas.
  • Sample (variance/stdev) divides by n-1; population (pvariance/pstdev) divides by n.
  • NormalDist is surprisingly powerful — CDF, inverse CDF, overlap, arithmetic on distributions.

graphlib.TopologicalSorter — Dependency Resolution

Algorithms & Data Structures

Dependency resolution is everywhere — build systems, task scheduling, package managers, database migrations. Python 3.9+ includes graphlib.TopologicalSorter for exactly this.

Example 1 — Basic topological sort

from graphlib import TopologicalSorter

# Task dependencies: {task: [depends_on...]}
ts = TopologicalSorter()
ts.add("cook", "prep_ingredients", "preheat_oven")
ts.add("prep_ingredients", "shop")
ts.add("preheat_oven")
ts.add("serve", "cook", "set_table")
ts.add("set_table")
ts.add("shop")

order = list(ts.static_order())
print(order)
# ['shop', 'preheat_oven', 'set_table', 'prep_ingredients', 'cook', 'serve']
['shop', 'preheat_oven', 'set_table', 'prep_ingredients', 'cook', 'serve']

Example 2 — From a dependency dict

from graphlib import TopologicalSorter

# Package dependencies

Example 4 — Detecting cycles

from graphlib import TopologicalSorter, CycleError

deps = {
    "A": ["B"],
    "B": ["C"],
    "C": ["A"],  # cycle!
}

ts = TopologicalSorter(deps)
try:
    list(ts.static_order())
except CycleError as e:
    print(f"Cycle detected: {e}")

Common Mistakes

Don’t
# get_ready() without prepare()
ts.get_ready()
# ValueError!

# Forgetting to call done()
ready = ts.get_ready()
# process tasks...
# sorter hangs — never progresses

# Adding nodes after prepare()
ts.prepare()
ts.add("new_task", "dep")
# ValueError!
Do
# Always call prepare() first
ts.prepare()
ready = ts.get_ready()

# Always mark nodes done
for task in ready:
    process(task)
    ts.done(task)

# Build entire graph before prepare()
for node, deps in all_deps.items():
    ts.add(node, *deps)
ts.prepare()

Key Rules

  • Use static_order() for simple sequential ordering. Use prepare()/get_ready()/done() when tasks can run in parallel.
  • CycleError is raised if the graph has cycles — catch it to give useful error messages.
  • The dict format is {node: [predecessors]} — the predecessors must be completed before the node.
